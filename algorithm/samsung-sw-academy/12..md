def findBrace(l):
	m = len(l)
	stack = ['']*1
	for idx in range(m):
		if l[idx] == '{' or l[idx]=='(':
			stack.append(l[idx])
		elif l[idx] == '}':
			if stack.pop() == '{': pass
			else: return 0
		elif l[idx] == ')':
			if stack.pop() == '(': pass
			else: return 0

	if len(stack)==1:
		return 1
	else: 
		return 0
            
T = int(input())

for test_case in range(1, T + 1):
	l = input()
	num = findBrace(l)
	print("#{} {}".format(test_case, num))
